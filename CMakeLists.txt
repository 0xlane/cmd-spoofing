# CMakeList.txt: cmd-spoofing 的 CMake 项目，在此处包括源代码并定义
# 项目特定的逻辑。
#
cmake_minimum_required (VERSION 3.8)

project ("cmd-spoofing")

if(MSVC)
    add_compile_options($<$<CONFIG:>:/MT> $<$<CONFIG:Debug>:/MTd> $<$<CONFIG:Release>:/MT>)
endif()

# 将源代码添加到此项目的可执行文件。
add_executable (cmd-spoofing "cmd-spoofing.cpp" "cmd-spoofing.h")
target_link_libraries(cmd-spoofing ntdll)

if (CMAKE_VERSION VERSION_GREATER 3.12)
  set_property(TARGET cmd-spoofing PROPERTY CXX_STANDARD 20)
endif()
